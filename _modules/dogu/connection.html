[docs]    def get_app_with_host(self, host):
        try:
            domain = host.split(':')[0]
            app = self.app_list.get(domain)
        except:
            app = None

        if app is None:  # can not find app by domain
            app = self.app_list.get('default')  # find default

            if app is None:
                raise ValueError('unknown domain')  # TODO catch error

        return app
